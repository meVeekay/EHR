	DROP SEQUENCE  "ADDRESS_SEQ";
	DROP SEQUENCE  "PATIENT_SEQ";
	DROP TABLE "PATIENT_ADDRESS";
	DROP TABLE "PATIENT";
  
  CREATE TABLE "PATIENT" 
   (	"PATIENT_ID" NUMBER(30,0) NOT NULL ENABLE, 
	"FIRST_NAME" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"LAST_NAME" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"DOB" DATE NOT NULL ENABLE, 
	"PRIMARY_PHONE" NUMBER, 
	"CELLPHONE" NUMBER, 
	"EMAIL" VARCHAR2(50 BYTE), 
	"SSN" VARCHAR2(20 BYTE), 
	"GENDER" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"EMERGENCY_CONTACTS" VARCHAR2(100 BYTE), 
	"PATIENT_ACC" NUMBER NOT NULL ENABLE, 
	 CONSTRAINT "PATIENT_PK" PRIMARY KEY ("PATIENT_ID")
   ) ;

  
  
  CREATE TABLE "PATIENT_ADDRESS" 
   (	"ADDRESS_ID" NUMBER(30,0) NOT NULL ENABLE, 
	"PATIENT_ID" NUMBER(30,0) NOT NULL ENABLE, 
	"STREET" VARCHAR2(50 BYTE), 
	"CITY" VARCHAR2(20 BYTE), 
	"STATE" VARCHAR2(20 BYTE), 
	"ZIP_CODE" NUMBER, 
	"ADDRESS_TYPE" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	 CONSTRAINT "PATIENT_ADDRESS_PK" PRIMARY KEY ("ADDRESS_ID"),   
	 CONSTRAINT "PATIENT_ADDRESS_FK1" FOREIGN KEY ("PATIENT_ID")
	  REFERENCES "PATIENT" ("PATIENT_ID") ENABLE
   ) ;
  
  CREATE SEQUENCE  "PATIENT_SEQ"  MINVALUE 10000 MAXVALUE 999999 INCREMENT BY 1 START WITH 10000 CACHE 20 NOORDER  NOCYCLE ;

  CREATE SEQUENCE  "ADDRESS_SEQ"  MINVALUE 1 MAXVALUE 10000 INCREMENT BY 1 START WITH 1 CACHE 20 NOORDER  NOCYCLE ;
